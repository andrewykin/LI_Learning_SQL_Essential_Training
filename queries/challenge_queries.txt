-- Challenge 1 | 1. How many transactions took place between 2011 - 2012?
SELECT
	InvoiceId,
	CustomerId,
	InvoiceDate,
	Total
FROM
	Invoice
WHERE 
	InvoiceDate >= '2011-01-01' AND InvoiceDate <= '2012-12-31'

-- OR

	COUNT(*)
FROM 
	Invoice
WHERE
	InvoiceDate >= '2011-01-01' AND InvoiceDate <= '2012-12-31'
